import numpy as np
import matplotlib.pyplot as plt

x = np.linspace(1, 50, 200)

# --------------------------------------------------
# Complejidad terica (sumas/restas)
# --------------------------------------------------
def eliminacion_gaussiana_complex(x):
    # ~ (1/3)n^3 + (1/2)n^2 - (5/6)n
    return x**3 / 3 + x**2 / 2 - 5 * x / 6


def gauss_jordan_complex(x):
    # ~ (1/2)n^3 - (1/2)n
    return x**3 / 2 - x / 2


# --------------------------------------------------
# Evaluacin
# --------------------------------------------------
y_gauss = eliminacion_gaussiana_complex(x)
y_gj = gauss_jordan_complex(x)

# --------------------------------------------------
# Grfica
# --------------------------------------------------
plt.figure(figsize=(8, 5))

plt.plot(x, y_gauss, label="Eliminacin Gaussiana", linewidth=2)
plt.plot(x, y_gj, label="Gauss-Jordan", linewidth=2)

plt.xlabel("n (nmero de incgnitas)")
plt.ylabel("Nmero de sumas/restas")
plt.title("Comparacin de complejidad: Gauss vs Gauss-Jordan")

plt.legend()
plt.grid(True)
plt.tight_layout()
plt.show()

import numpy as np
def eliminacion_gaussiana(A: np.ndarray | list[list[float | int]]) -> np.ndarray:
 
    sumas_restas = 0
    mult_div = 0
    intercambios = 0

    if not isinstance(A, np.ndarray):
        logging.debug("Convirtiendo A a numpy array.")
        A = np.array(A, dtype=float)

    assert A.shape[0] == A.shape[1] - 1, \
        "La matriz A debe ser de tamao n-by-(n+1)."

    n = A.shape[0]

    # ------------------------------------------------
    # Eliminacin hacia adelante
    # ------------------------------------------------
    for i in range(n - 1):
        # Buscar pivote
        p = None
        for pi in range(i, n):
            if A[pi, i] != 0:
                if p is None or abs(A[pi, i]) < abs(A[p, i]):
                    p = pi

        if p is None:
            raise ValueError("No existe solucin nica.")

        if p != i:
            A[[i, p], :] = A[[p, i], :]
            intercambios += 1

        for j in range(i + 1, n):
            m = A[j, i] / A[i, i]
            mult_div += 1
            for k in range(i, n + 1):
                A[j, k] = A[j, k] - m * A[i, k]
                mult_div += 1
                sumas_restas += 1

        logging.info(f"\n{A}")

    if A[n - 1, n - 1] == 0:
        raise ValueError("No existe solucin nica.")

    # ------------------------------------------------
    # Sustitucin hacia atrs
    # ------------------------------------------------
    solucion = np.zeros(n)

    solucion[n - 1] = A[n - 1, n] / A[n - 1, n - 1]
    mult_div += 1

    for i in range(n - 2, -1, -1):
        suma = 0
        for j in range(i + 1, n):
            suma += A[i, j] * solucion[j]
            mult_div += 1
            sumas_restas += 1
        solucion[i] = (A[i, n] - suma) / A[i, i]
        sumas_restas += 1
        mult_div += 1

    print("\nGauss:")
    print("Sumas/restas:\n",'expresin analtica:',n**3 / 3 + n**2  -   n / 3,
          '\nreal:', sumas_restas)
    print("Multiplicaciones/divisiones:\n",'expresin analtica:', n**3/3 + n**2 - n/3,
          '\nreal:', mult_div)
    print("Intercambios de fila:", intercambios)

    return solucion

import numpy as np
def gauss_jordan(A: np.ndarray | list[list[float | int]]) -> np.ndarray:
    """Resuelve un sistema de ecuaciones lineales mediante el mtodo de Gauss-Jordan
    con contadores de operaciones aritmticas.
    """
    sumas_restas = 0
    mult_div = 0

    if not isinstance(A, np.ndarray):
        logging.debug("Convirtiendo A a numpy array.")
        A = np.array(A, dtype=float)

    assert A.shape[0] == A.shape[1] - 1, \
        "La matriz A debe ser de tamao n-by-(n+1)."

    n = A.shape[0]

    for i in range(n):
        # ------------------------------------------------
        # Encontrar pivote
        # ------------------------------------------------
        p = None
        for pi in range(i, n):
            if A[pi, i] != 0:
                if p is None or abs(A[pi, i]) < abs(A[p, i]):
                    p = pi

        if p is None:
            raise ValueError("No existe solucin nica.")

        if p != i:
            A[[i, p], :] = A[[p, i], :]

        # ------------------------------------------------
        # Normalizar fila pivote
        # ------------------------------------------------
        pivote = A[i, i]
        for k in range(i, n + 1):
            A[i, k] = A[i, k] / pivote
            mult_div += 1

        # ------------------------------------------------
        # Eliminacin arriba y abajo
        # ------------------------------------------------
        for j in range(n):
            if j == i:
                continue
            m = A[j, i]
            for k in range(i, n + 1):
                A[j, k] = A[j, k] - m * A[i, k]
                mult_div += 1
                sumas_restas += 1

        logging.info(f"\n{A}")

    solucion = A[:, -1]

    print("\nGauss-Jordan:")
    print("Sumas/restas:\n",'expresin analtica:',n**3/2 - n/2,
          '\nreal:', sumas_restas)
    print("Multiplicaciones/divisiones:\n",'expresin analtica:', n**3/2 + n**2 - n/2,
          '\nreal:', mult_div)

    return solucion
