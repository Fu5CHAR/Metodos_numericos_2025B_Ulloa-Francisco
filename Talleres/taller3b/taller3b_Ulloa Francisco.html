import numpy as np

def seidel_modificado(A, b, X0, TOL=1e-6, N=100):
    """
    Mtodo de Gauss-Seidel para resolver sistemas lineales Ax = b
    
    Parmetros:
    A: matriz de coeficientes
    b: vector de trminos independientes
    X0: vector inicial
    TOL: tolerancia para la convergencia
    N: nmero mximo de iteraciones
    
    Retorna:
    Array con el historial de iteraciones o mensaje de error
    """
    
    A = np.array(A, dtype=float)
    b = np.array(b, dtype=float)
    X0 = np.array(X0, dtype=float)

    n = len(b)

    # Historial de iteraciones (iteracin 0)
    tabla = [X0.copy()]

    # Verificacin previa (iteraciones = 0)
    if np.all(np.abs(A @ X0 - b) < TOL):
        print('Nmero de iteraciones: 0')
        return np.array(tabla)

    k = 1
    x = X0.copy()  # Hacemos una copia para trabajar

    while k <= N:
        x_nuevo = x.copy()  # Copia del vector actual
        
        for i in range(n):
            # Suma para los elementos ya calculados en esta iteracin (j < i)
            suma1 = 0.0
            for j in range(i):
                suma1 += A[i, j] * x_nuevo[j]
            
            # Suma para los elementos an no actualizados (j > i)
            suma2 = 0.0
            for j in range(i + 1, n):
                suma2 += A[i, j] * x[j]
            
            # Actualizacin Gauss-Seidel
            x_nuevo[i] = (b[i] - suma1 - suma2) / A[i, i]

        tabla.append(x_nuevo.copy())

        # Condicin de convergencia
        if np.all(np.abs(x_nuevo - x) < TOL):
            print('Nmero de iteraciones (Seidel):', k)
            return np.array(tabla)

        x = x_nuevo.copy()  # Actualizamos para la siguiente iteracin
        k += 1

    print('Nmero mximo de iteraciones excedido (Seidel)')
    return "Nmero mximo de iteraciones excedido"

import numpy as np
import matplotlib.pyplot as plt

def jacobi(A, b, X0, TOL=1e-6, N=100):

    A = np.array(A, dtype=float)
    b = np.array(b, dtype=float)
    X0 = np.array(X0, dtype=float)

    n = len(b)
    k = 1

    # Historial de iteraciones
    tabla = [X0.copy()]

    while k <= N:
        x = np.zeros(n)

        # Paso 3
        for i in range(n):
            suma = 0.0
            for j in range(n):
                if j != i:
                    suma += A[i, j] * X0[j]

            x[i] = (b[i] - suma) / A[i, i]

        # Guardar iteracin completa
        tabla.append(x.copy())

        # Paso 4: condicin componente a componente
        if np.all(np.abs(x - X0) < TOL):
            tabla = np.array(tabla)

            print('Nmero de iteraciones:', k)
            print(tabla)
            return tabla

        # Paso 5 y 6
        X0 = x.copy()
        k += 1

    return "Nmero mximo de iteraciones excedido"

def seidel_modificado_grande(A, b, X0, TOL=1e-6, N=10000):  # 10,000 iteraciones!
    """
    Mtodo de Gauss-Seidel con ms iteraciones
    """
    
    A = np.array(A, dtype=float)
    b = np.array(b, dtype=float)
    x = np.array(X0, dtype=float)

    n = len(b)
    tabla = [x.copy()]

    # Verificacin previa
    if np.all(np.abs(A @ x - b) < TOL):
        print('Nmero de iteraciones (Seidel): 0')
        return np.array(tabla)

    k = 1
    while k <= N:
        x_nuevo = x.copy()
        
        for i in range(n):
            suma = np.dot(A[i, :i], x_nuevo[:i]) + np.dot(A[i, i+1:], x[i+1:])
            x_nuevo[i] = (b[i] - suma) / A[i, i]

        tabla.append(x_nuevo.copy())

        if np.all(np.abs(x_nuevo - x) < TOL):
            print('Nmero de iteraciones (Seidel):', k)
            return np.array(tabla)

        x = x_nuevo.copy()
        k += 1

    print(f'Nmero mximo de iteraciones excedido ({N} iteraciones)')
    # Devolvemos lo que tenemos hasta ahora
    return np.array(tabla)
